  <div class="card">

    <h1 style="margin:0 0 6px;">Live Interest Earned</h1>

    <div class="muted">This page displays <b>interest only</b> (principal is not shown). Updates every second.</div>


    <div class="grid" style="margin-top:12px;">

      <div>

        <label>APR (%)</label>

        <input id="apr" type="number" step="0.0001" value="5" />

      </div>

      <div>

        <label>Mode</label>

        <select id="mode">

          <option value="simple">Simple interest (linear)</option>

          <option value="continuous">Continuous compounding (smooth)</option>

        </select>

      </div>

    <label style="margin-top:12px;">Start time (when accrual begins)</label>

    <input id="start" type="datetime-local" />


    <!-- Principal is used for the calculation but hidden from display -->

    <label style="margin-top:12px;">Principal (hidden from viewers)</label>

    <input id="principal" type="password" step="0.01" value="100000" />


    <div class="row">

      <div class="muted">Interest accrued</div>

      <div id="interest" class="big mono">$0.00</div>


      <div class="muted" style="margin-top:12px;">Elapsed time: <span id="elapsed" class="mono"></span></div>

      <div class="muted" style="margin-top:6px;">Tip: you can “lock in” the principal by setting it once, then just share the link.</div>

    </div>

<script>

  const principalEl = document.getElementById('principal');

  const aprEl = document.getElementById('apr');

  const startEl = document.getElementById('start');

  const interestEl = document.getElementById('interest');

  const elapsedEl = document.getElementById('elapsed');

  const modeEl = document.getElementById('mode');


  // Default start time = now (local)

  const now = new Date();

  const pad = n => String(n).padStart(2,'0');

  const local = new Date(now.getTime() - now.getTimezoneOffset()*60000);

  startEl.value = `${local.getFullYear()}-${pad(local.getMonth()+1)}-${pad(local.getDate())}T${pad(local.getHours())}:${pad(local.getMinutes())}`;


  // Optional URL parameters (so you can share a pre-set link):

  // ?p=100000&apr=5&start=2026-02-11T20:00:00&mode=continuous

  const params = new URLSearchParams(location.search);

  if (params.has('p')) principalEl.value = params.get('p');

  if (params.has('apr')) aprEl.value = params.get('apr');

  if (params.has('mode')) modeEl.value = params.get('mode');

  if (params.has('start')) {

    const s = params.get('start').replace(' ', 'T');

    startEl.value = s.slice(0,16); // datetime-local expects YYYY-MM-DDTHH:MM

  }


  const fmt = new Intl.NumberFormat(undefined, { style:'currency', currency:'USD' });

  const SECONDS_PER_YEAR = 365.2425 * 24 * 60 * 60;


  function calc() {

    const P = Math.max(0, Number(principalEl.value || 0));     // hidden principal

    const r = Math.max(0, Number(aprEl.value || 0)) / 100;


    const start = new Date(startEl.value);

    const tSeconds = Math.max(0, (Date.now() - start.getTime()) / 1000);

    const tYears = tSeconds / SECONDS_PER_YEAR;


    let total;

    if (modeEl.value === 'continuous') total = P * Math.exp(r * tYears);

    else total = P * (1 + r * tYears);


    const interest = total - P;


    interestEl.textContent = fmt.format(interest);


    const hrs = Math.floor(tSeconds / 3600);

    const mins = Math.floor((tSeconds % 3600) / 60);

    const secs = Math.floor(tSeconds % 60);

    elapsedEl.textContent = `${hrs}h ${mins}m ${secs}s`;

  }


  setInterval(calc, 1000);

  calc();

</script>
